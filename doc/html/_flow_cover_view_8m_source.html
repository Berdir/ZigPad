00372 - (void)drawTile:(int)index atOffset:(double)off
00373 {
00374         FlowCoverRecord *fcr = [self getTileAtIndex:index];
00375         GLfloat m[16];
00376         memset(m,0,sizeof(m));
00377         m[10] = 1;
00378         m[15] = 1;
00379         m[0] = 1;
00380         m[5] = 1;
00381         double trans = off * SPREADIMAGE;
00382         
00383         double f = off * FLANKSPREAD;
00384         if (f < -FLANKSPREAD) {
00385                 f = -FLANKSPREAD;
00386         } else if (f > FLANKSPREAD) {
00387                 f = FLANKSPREAD;
00388         }
00389         m[3] = -f;
00390         m[0] = 1-fabs(f);
00391         double sc = 0.45 * (1 - fabs(f));
00392         trans += f * 1;
00393         
00394         glPushMatrix();
00395         glBindTexture(GL_TEXTURE_2D,fcr.texture);
00396         glTranslatef(trans, 0, 0);
00397         glScalef(sc,sc,1.0);
00398         glMultMatrixf(m);
00399         glDrawArrays(GL_TRIANGLE_STRIP,0,4);
00400         
00401         // reflect
00402         glTranslatef(0,-2,0);
00403         glScalef(1,-1,1);
00404         glColor4f(0.5,0.5,0.5,0.5);

00531          *      Adjust speed to make this land on an even location
