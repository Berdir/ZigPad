00275 - (void)observeValueForKeyPath:(NSString*)keyPath ofObject:(id)object change:(NSDictionary*)change context:(void*)context
00276 {
00277         if (context == BWOrderedChangeContext)
00278         {
00279                 //We listen for changes to the underlying relationship so we can update our ordering
00280                 //even for changes made through normal Core Data methods and not our ordered ones.
00281                 //Deleted objects are removed from the ordering, and newly added objects are simply
00282                 //added to the end of the ordering.
00283                 NSMutableArray*         newOrdering = [[self orderingForKey:keyPath] mutableCopy];
00284                 NSSet*                          removedObjects = [change objectForKey:NSKeyValueChangeOldKey];
00285                 NSSet*                          addedObjects = [change objectForKey:NSKeyValueChangeNewKey];
00286                 NSEnumerator*           enumerator;
00287                 NSManagedObject*        managedObject;
00288                 BOOL                            changedOrdering = NO;
00289 
00290                 //NSLog(@"%@ changed: removedObjects = %@, addedObjects = %@", object, removedObjects, addedObjects);
00291                 if ([removedObjects isEqual:[NSNull null]])
00292                         removedObjects = nil;
00293                 if ([addedObjects isEqual:[NSNull null]])
00294                         addedObjects = nil;
00295                 enumerator = [removedObjects objectEnumerator];
00296                 while (managedObject = [enumerator nextObject]) 
00297                 {
00298                         //If a set<Key>: method was called, it's possible for an object to be both in the
00299                         //addedObjects and removedObjects sets, in which case we just ignore it.
00300                         if (![addedObjects containsObject:managedObject])
00301                         {
00302                                 [newOrdering removeObject:[[managedObject objectID] URIRepresentation]];
00303                                 changedOrdering = YES;
00304                         }
00305                 }
00306                 enumerator = [addedObjects objectEnumerator];
00307                 while (managedObject = [enumerator nextObject])
00308                 {
00309                         //If a set<Key>: method was called, it's possible for an object to be both in the
00310                         //addedObjects and removedObjects sets, in which case we just ignore it.
00311                         //Also, we make sure not to add new objects a second time.
00312                         if (![removedObjects containsObject:managedObject] &&
00313                                 ![newOrdering containsObject:[[managedObject objectID] URIRepresentation]])
00314                         {
00315                                 [newOrdering addObject:[[managedObject objectID] URIRepresentation]];
00316                                 changedOrdering = YES;
00317                         }
00318                 }
00319                 //A small optimization to prevent unnecessary updates if nothing actually changed
00320                 if (changedOrdering)
00321                         [self setOrdering:newOrdering forKey:keyPath];
00322                 [newOrdering release];
00323         }
