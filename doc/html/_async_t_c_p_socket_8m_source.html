00205 - (unsigned)readLengthForTerm
00206 {
00207         NSAssert(term != nil, @"Searching for term in data when there is no term.");
00208         
00209         // What we're going to do is look for a partial sequence of the terminator at the end of the buffer.
00210         // If a partial sequence occurs, then we must assume the next bytes to arrive will be the rest of the term,
00211         // and we can only read that amount.
00212         // Otherwise, we're safe to read the entire length of the term.
00213         
00214         unsigned result = [term length];
00215         
00216         // Shortcut when term is a single byte
00217         if(result == 1) return result;
00218         
00219         // i = index within buffer at which to check data
00220         // j = length of term to check against
00221         
00222         // Note: Beware of implicit casting rules
00223         // This could give you -1: MAX(0, (0 - [term length] + 1));
00224         
00225         CFIndex i = MAX(0, (CFIndex)(bytesDone - [term length] + 1));
00226         CFIndex j = MIN([term length] - 1, bytesDone);
00227         
00228         while(i < bytesDone)
00229         {
00230                 const void *subBuffer = [buffer bytes] + i;
00231                 
00232                 if(memcmp(subBuffer, [term bytes], j) == 0)
00233                 {
00234                         result = [term length] - j;
00235                         break;
00236                 }
00237                 
00238                 i++;
00239                 j--;
00240         }
00241         
00242         if(maxLength > 0)
00243                 return MIN(result, (maxLength - bytesDone));
00244         else
00245                 return result;

00500 {
00501         // Check to make sure we're actually reading something right now
00502         if (!theCurrentRead) return NAN;
00503         
00504         // It's only possible to know the progress of our read if we're reading to a certain length
00505         // If we're reading to data, we of course have no idea when the data will arrive
00506         // If we're reading to timeout, then we have no idea when the next chunk of data will arrive.

00509         CFIndex d = theCurrentRead->bytesDone;
00510         CFIndex t = hasTotal ? [theCurrentRead->buffer length] : 0;
00511         if (tag != NULL)   *tag = theCurrentRead->tag;

00513         if (total != NULL) *total = t;

02377 - (void)doBytesAvailable
02378 {
02379         // If data is available on the stream, but there is no read request, then we don't need to process the data yet.
02380         // Also, if there is a read request, but no read stream setup yet, we can't process any data yet.
02381         if(theCurrentRead != nil && theReadStream != NULL)
02382         {
02383                 CFIndex totalBytesRead = 0;
02384                 
02385                 BOOL done = NO;
02386                 BOOL socketError = NO;
02387                 BOOL maxoutError = NO;
02388                 
02389                 while(!done && !socketError && !maxoutError && [self hasBytesAvailable])
02390                 {
02391                         BOOL didPreBuffer = NO;
02392                         
02393                         // If reading all available data, make sure there's room in the packet buffer.
02394                         if(theCurrentRead->readAllAvailableData == YES)
02395                         {
02396                                 // Make sure there is at least READALL_CHUNKSIZE bytes available.
02397                                 // We don't want to increase the buffer any more than this or we'll waste space.
02398                                 // With prebuffering it's possible to read in a small chunk on the first read.
02399                                 
02400                                 unsigned buffInc = READALL_CHUNKSIZE - ([theCurrentRead->buffer length] - theCurrentRead->bytesDone);
02401                                 [theCurrentRead->buffer increaseLengthBy:buffInc];
02402                         }
02403 
02404                         // If reading until data, we may only want to read a few bytes.
02405                         // Just enough to ensure we don't go past our term or over our max limit.
02406                         // Unless pre-buffering is enabled, in which case we may want to read in a larger chunk.
02407                         if(theCurrentRead->term != nil)
02408                         {
02409                                 // If we already have data pre-buffered, we obviously don't want to pre-buffer it again.
02410                                 // So in this case we'll just read as usual.
02411                                 
02412                                 if(([partialReadBuffer length] > 0) || !(theFlags & kEnablePreBuffering))
02413                                 {
02414                                         unsigned maxToRead = [theCurrentRead readLengthForTerm];
02415                                         
02416                                         unsigned bufInc = maxToRead - ([theCurrentRead->buffer length] - theCurrentRead->bytesDone);
02417                                         [theCurrentRead->buffer increaseLengthBy:bufInc];
02418                                 }
02419                                 else
02420                                 {
02421                                         didPreBuffer = YES;
02422                                         unsigned maxToRead = [theCurrentRead prebufferReadLengthForTerm];
02423                                         
02424                                         unsigned buffInc = maxToRead - ([theCurrentRead->buffer length] - theCurrentRead->bytesDone);
02425                                         [theCurrentRead->buffer increaseLengthBy:buffInc];
02426 
02427                                 }
02428                         }
02429                         
02430                         // Number of bytes to read is space left in packet buffer.
02431                         CFIndex bytesToRead = [theCurrentRead->buffer length] - theCurrentRead->bytesDone;
02432

02446

02449                         }
02450

02452                         if(theCurrentRead->readAllAvailableData != YES)
02453                         {
02454                                 if(theCurrentRead->term != nil)

02456                                         if(didPreBuffer)
02457                                         {

02466                                                         
02467                                                         [partialReadBuffer appendBytes:overflowBuffer length:overflow];

02469                                                         // Update the bytesDone variable.
02470                                                         // Note: The completeCurrentRead method will trim the buffer for us.
02471                                                         theCurrentRead->bytesDone -= overflow;
02472                                                 }
02473                                                 
02474                                                 done = (overflow >= 0);
02475                                         }
02476                                         else
02477                                         {
02478                                                 // Search for the terminating sequence at the end of the buffer
02479                                                 int termlen = [theCurrentRead->term length];
02480                                                 if(theCurrentRead->bytesDone >= termlen)
02481                                                 {
02482                                                         const void *buf = [theCurrentRead->buffer bytes] + (theCurrentRead->bytesDone - termlen);
02483                                                         const void *seq = [theCurrentRead->term bytes];
02484                                                         done = (memcmp (buf, seq, termlen) == 0);
02485                                                 }
02486                                         }
02487                                         
02488                                         if(!done && theCurrentRead->maxLength >= 0 && theCurrentRead->bytesDone >= theCurrentRead->maxLength)
02489                                         {
02490                                                 // There's a set maxLength, and we've reached that maxLength without completing the read
02491                                                 maxoutError = YES;
02492                                         }
02493                                 }
02494                                 else
02495                                 {
02496                                         // Done when (sized) buffer is full.
02497                                         done = ([theCurrentRead->buffer length] == theCurrentRead->bytesDone);
02498                                 }
02499                         }
02500                         // else readAllAvailable doesn't end until all readable is read.
02501                 }
02502                 
02503                 if(theCurrentRead->readAllAvailableData && theCurrentRead->bytesDone > 0)
02504                         done = YES;     // Ran out of bytes, so the "read-all-data" type packet is done
02505 
02506                 if(done)

02508                         [self completeCurrentRead];
02509                         if (!socketError) [self scheduleDequeueRead];
