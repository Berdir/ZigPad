00085 static void TCPServerAcceptCallBack(CFSocketRef socket, CFSocketCallBackType type, CFDataRef address, const void *data, void *info) {
00086     TCPServer *server = (TCPServer *)info;
00087     if (kCFSocketAcceptCallBack == type) { 
00088         // for an AcceptCallBack, the data parameter is a pointer to a CFSocketNativeHandle
00089         CFSocketNativeHandle nativeSocketHandle = *(CFSocketNativeHandle *)data;
00090         uint8_t name[SOCK_MAXADDRLEN];
00091         socklen_t namelen = sizeof(name);
00092         NSData *peer = nil;
00093         if (0 == getpeername(nativeSocketHandle, (struct sockaddr *)name, &namelen)) {
00094             peer = [NSData dataWithBytes:name length:namelen];
00095         }
00096         CFReadStreamRef readStream = NULL;
00097                 CFWriteStreamRef writeStream = NULL;
00098         CFStreamCreatePairWithSocket(kCFAllocatorDefault, nativeSocketHandle, &readStream, &writeStream);
00099         if (readStream && writeStream) {
00100             CFReadStreamSetProperty(readStream, kCFStreamPropertyShouldCloseNativeSocket, kCFBooleanTrue);
00101             CFWriteStreamSetProperty(writeStream, kCFStreamPropertyShouldCloseNativeSocket, kCFBooleanTrue);
00102             [server handleNewConnectionFromAddress:peer inputStream:(NSInputStream *)readStream outputStream:(NSOutputStream *)writeStream];
00103         } else {
00104             // on any failure, need to destroy the CFSocketNativeHandle 
00105             // since we are not going to use it any more
00106             close(nativeSocketHandle);
00107         }
00108         if (readStream) CFRelease(readStream);
00109         if (writeStream) CFRelease(writeStream);
