00044 - (void) receiveSyncEvent: (NSNotification *) notification {
00045     SyncEvent *event = [notification object];
00046     
00047     switch (event.command) {
00048         case ANSWER:
00049         case SELECT:
00050         {
00051             //get Presentation with selected refId from Database.
00052             NSManagedObjectContext* context =  [[Database sharedInstance] managedObjectContext];
00053             NSEntityDescription *entityDescription = [NSEntityDescription entityForName:@"Presentation" inManagedObjectContext:context];
00054             NSFetchRequest *request = [[NSFetchRequest alloc] init] ;
00055             [request setEntity:entityDescription];
00056             
00057             // Add a corresponding filter.
00058             NSPredicate *filter = [NSPredicate predicateWithFormat:[NSString stringWithFormat:@"refId = %d", event.argument]];
00059             [request setPredicate:filter];
00060             
00061             // Read db
00062             NSError* error = nil;
00063             NSArray *presentations = [context executeFetchRequest:request error:&error];
00064             
00065             if ([presentations count] == 0) {
00066                 NSLog(@"No presentation for the given refId (%d) found", event.argument);
00067                 [request release];
00068                 return;
00069             }
00070             
00071             self.activePresentation = [presentations objectAtIndex:0];
00072             [request release];
00073             
00074             NSLog(@"Selected Presentation through sync event %@.", self.activePresentation.name);
00075             
00076             // Only jump to the first action of this presentation if this is a SELECT
00077             // command.
00078             if (event.command == SELECT) { 
00079                 [self.activePresentation resetPresentation];
00080                 Action *a = [self.activePresentation getNextAction];    
00081                 ActionViewController *nextPage = [ActionViewController getViewControllerFromAction:a];
00082                 nextPage.presentation = self.activePresentation;
00083                 
00084                 [AnimatorHelper slideWithAnimation:-1 :self :nextPage :false :true :true];
00085             }
00086             break;
00087         }
00088         case JUMP:
00089             // Only act on jump commands if no presentation is active and a presentation was
00090             // already selected with the ANSWER command.
00091             if ([self.navigationController.viewControllers count] == 1 && self.activePresentation) {
00092                 Action *a = [self.activePresentation jumpToAction:event.argument_lowerByte sequenceIndex:event.argument_upperByte];
00093             
00094                 if (a) {
00095                     ActionViewController *nextPage = [ActionViewController getViewControllerFromAction:a];
00096                     nextPage.presentation = self.activePresentation;
00097                     [AnimatorHelper slideWithAnimation:-1 :self :nextPage :false :true :true];
00098                 }
00099             }
00100             break;
00101         case CONNECTED:
00102             self.navigationController.navigationBar.topItem.title = @"ZigPad (Synced)";
00103             
00104             // Send a REQUEST if no presentation is running.
00105             if ([self.navigationController.viewControllers count] == 1) {
00106                 SyncEvent *event = [[SyncEvent alloc] init];
00107                 event.command = REQUEST;
00108                 
00109                 [[NSNotificationCenter defaultCenter] postNotificationName:@"ZigPadSyncFire" object:event];
00110                 
00111                 [event release];
00112             }
00113             break;
00114         case LOST_CONNECTION:
00115             self.navigationController.navigationBar.topItem.title = @"ZigPad";
00116             break;
00117     }
